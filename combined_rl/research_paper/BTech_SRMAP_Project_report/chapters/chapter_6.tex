\chapter{HARDWARE / SOFTWARE TOOLS USED}
%-------------------------------------------------------------
\section{Hardware Requirements}
The hardware components used for the implementation 
of the proposed system are listed below.
\subsection{Processor}
\begin{itemize}
\item Processor : 11th Gen Intel(R) Core(TM) i5-11320H  
\item Base Speed : 2.50 GHz  
\item Max Speed : 3.20 GHz  
\item Architecture : 64-bit, x64-based  
\end{itemize}
\subsection{Memory (RAM)}
\begin{itemize}
\item Installed RAM : 16.0 GB 
\end{itemize}
\subsection{Storage}
\begin{itemize}
\item Storage Type : HDD / SSD 
\end{itemize}
\subsection{Network Interface}
\begin{itemize}
\item Ethernet or Wi-Fi Adapter  
\end{itemize}
%-------------------------------------------------------------
\section{Software Requirements}
The software tools used for implementation 
of the proposed system are listed below.
\subsection{Operating System}
\begin{itemize}
\item Linux-based development environment
\end{itemize}
\subsection{Programming Language}
\begin{itemize}
\item Python  
\end{itemize}
\subsection{Machine Learning Library}
\begin{itemize}
\item Scikit-learn  
\item NumPy  
\item Pandas  
\end{itemize}
\subsection{Reinforcement Learning and Simulation Stack}
\begin{itemize}
\item Closed-loop RL evaluation pipeline for scheduling and aggregation
\item ns-3-generated traffic traces used as the evaluation dataset
\item Q-table based policy execution for the combined model
\item Matplotlib for producing comparison graphs and summary plots
\end{itemize}
\subsection{Development Tools}
\begin{itemize}
\item Python virtual environment for reproducible execution  
\item Spyder IDE and command-line execution for experimentation  
\end{itemize}
\subsection{Visualization Tools}
\begin{itemize}
\item Matplotlib
\end{itemize}
